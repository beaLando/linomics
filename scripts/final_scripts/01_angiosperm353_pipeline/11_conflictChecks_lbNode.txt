    taxa_to_remove = setdiff(tipLabels(t), keep_taxa_noM)
    for taxon in taxa_to_remove
        deleteleaf!(t, taxon)
    end

    # 2. Check for sufficient tree size
    # A tree needs at least 3 tips to be meaningfully rooted
    if length(tipLabels(t)) >= 3
        
        # Sync the internal structure after pruning
        directEdges!(t)
    
    # 3. Root safely
    if "HNCF_nuclear" in tipLabels(t)
        rootatnode!(t, "HNCF_nuclear")
    elseif "ERR2040380_nuclear" in tipLabels(t)
        rootatnode!(t, "ERR2040380_nuclear")
    end
else
        @warn "Tree skipped: fewer than 3 taxa remaining after pruning."
end
end

# note: the following produces a file with trees listed correctly and the InferNetwork command set up as a canva. I then opened it and changed parameters manually if I needed to.
function write_phylonet_nexus(trees, filename; hmax=0, nruns=10)
    open(filename, "w") do io
        write(io, "#NEXUS\n\n")
        
        # BEGIN TREES block
        write(io, "BEGIN TREES;\n")
        for (i, t) in enumerate(trees)
            # Write each tree in Newick format
            write(io, "Tree gt$i = $(writeTopology(t))\n")
        end
        write(io, "END;\n\n")
        
        # BEGIN PHYLONET block
        write(io, "BEGIN PHYLONET;\n")
        
        # We use InferNetwork_ML for Maximum Likelihood
        # -maxh: Maximum number of reticulations
        # -n: Number of optimal networks to return
        # -pl: Number of processors (adjust as needed)
        # -mr: Number of runs for the search heuristic
        tree_names = ["gt$i" for i in 1:length(trees)]
        trees_str = join(tree_names, " ")
        
        write(io, "InferNetwork_MP (all) $hmax \n")
        write(io, " -n 10 \n")
        write(io, " -x 10 \n")
        write(io, " -pl 6 \n") # Uses 4 cores
        write(io, " ;\n")
        
        write(io, "END;\n")
    end
    println("PhyloNet NEXUS file saved to: $filename")
end
